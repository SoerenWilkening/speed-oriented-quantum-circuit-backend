**1. CLA vs RCA Equivalence Tests (Phase SC #2):**

For widths 1-6, compare CLA and RCA results for all input pairs (QQ variant). Use `ql.option('cla', True)` vs `ql.option('cla', False)` to switch.

```python
"""Phase 71: CLA Verification - Comprehensive Tests.

Validates all phase success criteria:
SC1: QQ_add_cla computes addition using generate/propagate prefix tree
SC2: CLA produces identical results to RCA for all input pairs at widths 1-6
SC3: CLA circuit depth < RCA depth for widths >= 8
SC4: All ancilla correctly uncomputed and freed
"""

import gc
import warnings
import pytest
from qiskit_aer import AerSimulator
import qiskit.qasm3
from verify_helpers import format_failure_message
import quantum_language as ql

warnings.filterwarnings("ignore", message="Value .* exceeds")


def _simulate_and_extract(qasm_str, num_qubits, result_start, result_width):
    """Simulate QASM and extract result register value."""
    circuit = qiskit.qasm3.loads(qasm_str)
    if not circuit.cregs:
        circuit.measure_all()
    simulator = AerSimulator(method="statevector")
    job = simulator.run(circuit, shots=1)
    result = job.result()
    counts = result.get_counts()
    bitstring = list(counts.keys())[0]
    msb_pos = num_qubits - result_start - result_width
    lsb_pos = num_qubits - 1 - result_start
    result_bits = bitstring[msb_pos : lsb_pos + 1]
    return int(result_bits, 2)


class TestCLAvsRCAEquivalence:
    """SC2: CLA and RCA produce identical results for all input pairs."""

    @pytest.mark.parametrize("width", [1, 2, 3, 4, 5, 6])
    def test_qq_add_equivalence(self, width):
        """QQ addition: CLA == RCA for every input pair."""
        modulus = 1 << width
        failures = []

        for a_val in range(modulus):
            for b_val in range(modulus):
                # RCA result
                ql.option("fault_tolerant", True)
                ql.option("cla", False)  # Force RCA
                ql.init(2 * width)
                qa_rca = ql.qint(width, a_val)
                qb_rca = ql.qint(width, b_val)
                qa_rca += qb_rca
                qasm_rca = ql.qasm()
                nq_rca = ql._get_num_qubits()
                rca_result = _simulate_and_extract(qasm_rca, nq_rca, 0, width)

                # CLA result
                ql.option("cla", True)  # Enable CLA
                ql.init(2 * width)
                qa_cla = ql.qint(width, a_val)
                qb_cla = ql.qint(width, b_val)
                qa_cla += qb_cla
                qasm_cla = ql.qasm()
                nq_cla = ql._get_num_qubits()
                cla_result = _simulate_and_extract(qasm_cla, nq_cla, 0, width)

                if rca_result != cla_result:
                    failures.append(
                        f"  {a_val}+{b_val}: RCA={rca_result}, CLA={cla_result}"
                    )
                gc.collect()

        assert not failures, format_failure_message(
            f"QQ add CLA vs RCA width={width}", failures
        )

    @pytest.mark.parametrize("width", [1, 2, 3, 4, 5])
    def test_cq_add_equivalence(self, width):
        """CQ addition: CLA == RCA for every input pair."""
        modulus = 1 << width
        failures = []

        for val in range(modulus):
            for self_val in range(modulus):
                # RCA
                ql.option("fault_tolerant", True)
                ql.option("cla", False)
                ql.init(width)
                qa_rca = ql.qint(width, self_val)
                qa_rca += val
                qasm_rca = ql.qasm()
                nq_rca = ql._get_num_qubits()
                rca_result = _simulate_and_extract(qasm_rca, nq_rca, 0, width)

                # CLA
                ql.option("cla", True)
                ql.init(width)
                qa_cla = ql.qint(width, self_val)
                qa_cla += val
                qasm_cla = ql.qasm()
                nq_cla = ql._get_num_qubits()
                cla_result = _simulate_and_extract(qasm_cla, nq_cla, 0, width)

                if rca_result != cla_result:
                    failures.append(
                        f"  {self_val}+{val}: RCA={rca_result}, CLA={cla_result}"
                    )
                gc.collect()

        assert not failures, format_failure_message(
            f"CQ add CLA vs RCA width={width}", failures
        )

    @pytest.mark.parametrize("width", [4, 5])
    def test_qq_sub_equivalence(self, width):
        """QQ subtraction: CLA == RCA for every input pair."""
        modulus = 1 << width
        failures = []
        for a_val in range(modulus):
            for b_val in range(modulus):
                ql.option("fault_tolerant", True)
                ql.option("cla", False)
                ql.init(2 * width)
                qa = ql.qint(width, a_val)
                qb = ql.qint(width, b_val)
                qa -= qb
                rca_result = _simulate_and_extract(ql.qasm(), ql._get_num_qubits(), 0, width)

                ql.option("cla", True)
                ql.init(2 * width)
                qa = ql.qint(width, a_val)
                qb = ql.qint(width, b_val)
                qa -= qb
                cla_result = _simulate_and_extract(ql.qasm(), ql._get_num_qubits(), 0, width)

                if rca_result != cla_result:
                    failures.append(f"  {a_val}-{b_val}: RCA={rca_result}, CLA={cla_result}")
                gc.collect()
        assert not failures, format_failure_message(f"QQ sub CLA vs RCA width={width}", failures)
```

```python
class TestCLAAncillaCleanup:
    """SC4: All ancilla qubits uncomputed to |0> after CLA."""

    @pytest.mark.parametrize("width", [4, 5])
    @pytest.mark.parametrize("variant", ["bk", "ks"])
    def test_qq_ancilla_cleanup(self, width, variant):
        """After QQ CLA, all ancilla qubits should be |0>."""
        ql.option("fault_tolerant", True)
        ql.option("cla", True)
        ql.option("qubit_saving", variant == "bk")

        # Use specific input values that exercise the full adder
        test_pairs = [(1, 1), (3, 5), ((1 << width) - 1, 1)]
        for a_val, b_val in test_pairs:
            a_val = a_val % (1 << width)
            b_val = b_val % (1 << width)
            ql.init(2 * width)
            qa = ql.qint(width, a_val)
            qb = ql.qint(width, b_val)
            qa += qb
            qasm_str = ql.qasm()
            num_qubits = ql._get_num_qubits()

            # Full statevector simulation
            circuit = qiskit.qasm3.loads(qasm_str)
            circuit.save_statevector()
            sim = AerSimulator(method="statevector")
            job = sim.run(circuit, shots=1)
            sv = job.result().get_statevector()
            probs = sv.probabilities()

            # Check: only states where ancilla qubits are |0> should have nonzero probability
            # Data qubits: 0..2*width-1 (qa and qb)
            # Ancilla qubits: 2*width..num_qubits-1
            data_qubits = 2 * width
            for state_idx, prob in enumerate(probs):
                if prob > 1e-10:
                    # Extract ancilla bits
                    ancilla_val = state_idx >> data_qubits
                    assert ancilla_val == 0, (
                        f"{variant.upper()} QQ w={width}: {a_val}+{b_val}: "
                        f"state {state_idx:0{num_qubits}b} has ancilla={ancilla_val:#x} "
                        f"(expected 0), prob={prob:.6f}"
                    )
            gc.collect()

    @pytest.mark.parametrize("width", [4, 5])
    @pytest.mark.parametrize("variant", ["bk", "ks"])
    def test_cq_ancilla_cleanup(self, width, variant):
        """After CQ CLA, all ancilla (temp + CLA) should be |0>."""
        ql.option("fault_tolerant", True)
        ql.option("cla", True)
        ql.option("qubit_saving", variant == "bk")

        test_values = [1, 3, (1 << width) - 1]
        for val in test_values:
            val = val % (1 << width)
            for self_val in [0, 1, (1 << width) - 1]:
                self_val = self_val % (1 << width)
                ql.init(width)
                qa = ql.qint(width, self_val)
                qa += val
                qasm_str = ql.qasm()
                num_qubits = ql._get_num_qubits()

                circuit = qiskit.qasm3.loads(qasm_str)
                circuit.save_statevector()
                sim = AerSimulator(method="statevector")
                job = sim.run(circuit, shots=1)
                sv = job.result().get_statevector()
                probs = sv.probabilities()

                data_qubits = width  # only self register is data
                for state_idx, prob in enumerate(probs):
                    if prob > 1e-10:
```
