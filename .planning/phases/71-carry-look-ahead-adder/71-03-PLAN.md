**1. Implement controlled BK/KS QQ adders:**

`toffoli_cQQ_add_bk(int bits)` and `toffoli_cQQ_add_ks(int bits)`:

Follow the same pattern as existing `toffoli_cQQ_add()` (controlled CDKM): replace every gate with its controlled version.

- Every `ccx(target, ctrl1, ctrl2)` becomes `mcx(target, [ctrl1, ctrl2, ext_ctrl], 3)` (3 controls)
- Every `cx(target, ctrl)` becomes `ccx(target, ctrl, ext_ctrl)` (2 controls)
- Every `x(target)` becomes `cx(target, ext_ctrl)` (controlled NOT)

**Qubit layout for cQQ BK:**
```
[0..bits-1]        = register a (source, preserved)
[bits..2*bits-1]   = register b (target, gets a+b)
[2*bits..4*bits-3] = CLA ancilla (2*(bits-1))
[4*bits-2]         = external control qubit
Total: 4*bits - 1 qubits
```

**Qubit layout for cQQ KS:**
```
[0..bits-1]        = register a (source, preserved)
[bits..2*bits-1]   = register b (target, gets a+b)
[2*bits..]         = CLA ancilla (ks_ancilla_count)
[2*bits+ks_anc]    = external control qubit
Total: 2*bits + ks_ancilla_count(bits) + 1 qubits
```

Cache controlled BK/KS sequences (separate cache arrays: precompiled_toffoli_cQQ_add_bk[65], precompiled_toffoli_cQQ_add_ks[65]).

If the BK/KS implementations from Plans 01/02 use internal static helpers `_emit_bk_tree()` / `_emit_ks_tree()`, create controlled versions `_emit_cbk_tree()` / `_emit_cks_tree()` that take an `ext_ctrl` parameter and emit controlled gates. This keeps the code DRY without violating the "inline/self-contained per variant" constraint.

Alternatively, if the tree helpers are not factored out, duplicate the tree logic with controlled gate substitutions. This is more verbose but follows the exact pattern of emit_cMAJ/emit_cUMA in the existing code.

**2. Implement controlled BK/KS CQ adders:**

`toffoli_cCQ_add_bk(int bits, int64_t value)` and `toffoli_cCQ_add_ks(int bits, int64_t value)`:

Follow the same temp-register pattern as existing `toffoli_cCQ_add()`:
1. CX-init temp register (controlled by ext_ctrl): `cx(temp[i], ext_ctrl)` for set bits
2. Controlled CLA adder core (same as cQQ BK/KS tree)
3. CX-cleanup temp register: `cx(temp[i], ext_ctrl)` for set bits

**Qubit layout for cCQ BK:**
```
[0..bits-1]        = temp register (controlled init)
[bits..2*bits-1]   = self register (target)
[2*bits..4*bits-3] = CLA ancilla (2*(bits-1))
[4*bits-2]         = external control qubit
Total: 4*bits - 1 qubits
```

**Qubit layout for cCQ KS:**
```
[0..bits-1]        = temp register
[bits..2*bits-1]   = self register (target)
[2*bits..]         = CLA ancilla (ks_ancilla_count)
[2*bits+ks_anc]    = external control qubit
Total: 2*bits + ks_ancilla_count(bits) + 1 qubits
```

NOT cached (value-dependent). Caller must free via toffoli_sequence_free().

if (controlled) {
    /* Check CLA eligibility for controlled path */
    if (circ->cla_override == 0 && result_bits >= CLA_THRESHOLD) {
        int cla_ancilla_count;
        if (circ->qubit_saving) {
            cla_ancilla_count = 2 * (result_bits - 1);
        } else {
            cla_ancilla_count = ks_ancilla_count(result_bits);

        qubit_t cla_ancilla = allocator_alloc(circ->allocator, cla_ancilla_count, true);
        if (cla_ancilla != (qubit_t)-1) {
            unsigned int cla_qa[256];
            /* a-register [0..bits-1] = other */
            for (i = 0; i < other_bits; i++) cla_qa[i] = other_qubits[i];
            /* b-register [bits..2*bits-1] = self */
            for (i = 0; i < self_bits; i++) cla_qa[result_bits + i] = self_qubits[i];

            for (i = 0; i < cla_ancilla_count; i++)

**3. Add controlled CLA tests to `tests/python/test_cla_addition.py`:**

Test controlled CLA addition using the `with` block pattern (same as Phase 67 tests):

```python
class TestControlledCLAAddition:
    """Test controlled CLA variants (cQQ and cCQ)."""

    @pytest.mark.parametrize("variant", ["bk", "ks"])
    @pytest.mark.parametrize("width", [4, 5])
    def test_cqq_cla_add(self, variant, width):
        """Controlled QQ addition: a += b when control is |1>."""
        ql.option("fault_tolerant", True)
        ql.option("cla", True)
        ql.option("qubit_saving", variant == "bk")
        modulus = 1 << width
        failures = []

        # Test with control = 1 (addition should happen)
        for a_val in range(min(modulus, 8)):  # Sample for speed
            for b_val in range(min(modulus, 8)):
                expected = (a_val + b_val) % modulus
                ql.init(2 * width + 1)
                qa = ql.qint(width, a_val)
                qb = ql.qint(width, b_val)
                ctrl = ql.qint(1, 1)  # control = |1>
                with ctrl:
                    qa += qb
                qasm_str = ql.qasm()
                num_qubits = ql._get_num_qubits()
                actual = _simulate_and_extract(qasm_str, num_qubits, 0, width)
                if actual != expected:
                    failures.append(f"  ctrl=1: {a_val} + {b_val}: expected {expected}, got {actual}")
                gc.collect()

        # Test with control = 0 (no addition)
        for a_val in range(min(modulus, 4)):
            for b_val in range(min(modulus, 4)):
                ql.init(2 * width + 1)
                qa = ql.qint(width, a_val)
                qb = ql.qint(width, b_val)
                ctrl = ql.qint(1, 0)  # control = |0>
                with ctrl:
                    qa += qb
                qasm_str = ql.qasm()
                num_qubits = ql._get_num_qubits()
                actual = _simulate_and_extract(qasm_str, num_qubits, 0, width)
                if actual != a_val:  # Should be unchanged
                    failures.append(f"  ctrl=0: {a_val} + {b_val}: expected {a_val} (unchanged), got {actual}")
                gc.collect()

        assert not failures, format_failure_message(
            f"cQQ CLA {variant.upper()} add width={width}", failures
        )

    @pytest.mark.parametrize("variant", ["bk", "ks"])
    @pytest.mark.parametrize("width", [4, 5])
    def test_ccq_cla_add(self, variant, width):
        """Controlled CQ addition: a += classical when control is |1>."""
        ql.option("fault_tolerant", True)
        ql.option("cla", True)
        ql.option("qubit_saving", variant == "bk")
        modulus = 1 << width
        failures = []

        for val in range(min(modulus, 8)):
            for self_val in range(min(modulus, 8)):
                expected = (self_val + val) % modulus
                ql.init(width + 1)
                qa = ql.qint(width, self_val)
                ctrl = ql.qint(1, 1)
                with ctrl:
                    qa += val
                qasm_str = ql.qasm()
                num_qubits = ql._get_num_qubits()
                actual = _simulate_and_extract(qasm_str, num_qubits, 0, width)
                if actual != expected:
                    failures.append(f"  ctrl=1: {self_val} + {val}: expected {expected}, got {actual}")
                gc.collect()

        # Control = 0: no change
        for self_val in range(min(modulus, 4)):
            ql.init(width + 1)
            qa = ql.qint(width, self_val)
            ctrl = ql.qint(1, 0)
            with ctrl:
                qa += 3
            qasm_str = ql.qasm()
            num_qubits = ql._get_num_qubits()
            actual = _simulate_and_extract(qasm_str, num_qubits, 0, width)
            if actual != self_val:
                failures.append(f"  ctrl=0: {self_val} + 3: expected {self_val}, got {actual}")
            gc.collect()

        assert not failures, format_failure_message(
            f"cCQ CLA {variant.upper()} add width={width}", failures
        )

    @pytest.mark.parametrize("variant", ["bk", "ks"])
    def test_controlled_cla_subtraction(self, variant):
        """Controlled CLA subtraction works via inversion."""
        ql.option("fault_tolerant", True)
        ql.option("cla", True)
        ql.option("qubit_saving", variant == "bk")
        width = 4
        modulus = 16
        failures = []

        for a_val in range(min(modulus, 8)):
            for b_val in range(min(modulus, 8)):
                expected = (a_val - b_val) % modulus
                ql.init(2 * width + 1)
                qa = ql.qint(width, a_val)
                qb = ql.qint(width, b_val)
                ctrl = ql.qint(1, 1)
                with ctrl:
                    qa -= qb
                qasm_str = ql.qasm()
                num_qubits = ql._get_num_qubits()
                actual = _simulate_and_extract(qasm_str, num_qubits, 0, width)
                if actual != expected:
                    failures.append(f"  ctrl=1: {a_val} - {b_val}: expected {expected}, got {actual}")
                gc.collect()

        assert not failures, format_failure_message(
            f"Controlled CLA {variant.upper()} sub width=4", failures
        )
```
