Eliminate global state and document memory ownership in C backend

Purpose: Global state (`circuit`, `QPU_state`, `instruction_list`) makes the code hard to test, thread-unsafe, and error-prone. Functions that modify circuit state should accept explicit circuit_t* parameter. Memory ownership documentation enables future cleanup and prevents leaks.

Output: No global circuit variable; all circuit-modifying functions accept circuit_t* parameter; memory ownership documented at every allocation; Python bindings updated.
</objective>

```c
quantum_int_t *QBOOL(circuit_t *circ) {
    // OWNERSHIP: Caller owns returned quantum_int_t*, must free when done
    if (circ == NULL) return NULL;
    quantum_int_t *integer = malloc(sizeof(quantum_int_t));
    if (integer == NULL) return NULL;

    integer->MSB = INTEGERSIZE - 1;
    integer->q_address[INTEGERSIZE - 1] = circ->used_qubit_indices;

    circ->ancilla += 1;
    circ->used_qubit_indices += 1;
    return integer;
}
```

QINT():
```c
quantum_int_t *QINT(circuit_t *circ) {
    // OWNERSHIP: Caller owns returned quantum_int_t*, must free when done
    if (circ == NULL) return NULL;
    quantum_int_t *integer = malloc(sizeof(quantum_int_t));
    if (integer == NULL) return NULL;

    integer->MSB = 0;
    circ->ancilla += INTEGERSIZE;

    for (int i = 0; i < INTEGERSIZE; ++i) {
        integer->q_address[i] = circ->used_qubit_indices;
        circ->used_qubit_indices++;
    }
    return integer;
}
```

free_element():

```c
void free_element(circuit_t *circ, quantum_int_t *el1) {
    // OWNERSHIP: Frees the quantum_int_t, updates circuit tracking
    if (circ == NULL || el1 == NULL) return;
    circ->ancilla--;
    circ->used_qubit_indices--;
    free(el1);  // Actually free the element
}
```
