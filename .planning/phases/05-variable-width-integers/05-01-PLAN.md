```c
quantum_int_t *bit_of_int(quantum_int_t *el1, int bit) {
    quantum_int_t *b = malloc(sizeof(quantum_int_t));
    if (b == NULL) {
        return NULL;
    }
    b->width = 1;
    b->MSB = 63;  // Last element in right-aligned array
    b->q_address[63] = el1->q_address[64 - el1->width + bit];
    return b;
}
```
