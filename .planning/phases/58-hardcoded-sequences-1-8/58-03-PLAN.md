"""Validation tests for hardcoded addition sequences.

These tests verify that hardcoded (pre-computed) sequences for widths 1-8
produce identical results to the dynamic generation path.

Per CONTEXT.md, this is one-time verification (not in regular CI).
Mark with @pytest.mark.hardcoded_validation to allow filtering.
"""

import pytest
import quantum_language as ql


@pytest.mark.hardcoded_validation
class TestHardcodedSequenceValidation:
    """Verify hardcoded sequences match dynamic generation."""

    @pytest.mark.parametrize("width", [1, 2, 3, 4, 5, 6, 7, 8])
    def test_qq_add_produces_correct_results(self, verify_circuit, width):
        """QQ_add hardcoded produces correct arithmetic results."""
        # Test a few representative values
        test_cases = [(0, 0), (1, 1), (2, 3)]
        max_val = (1 << width) - 1
        test_cases.extend([(max_val, 1), (max_val, max_val)])

        for a, b in test_cases:
            if a >= (1 << width) or b >= (1 << width):
                continue

            def circuit_builder(a=a, b=b, w=width):
                qa = ql.qint(a, width=w)
                qb = ql.qint(b, width=w)
                _r = qa + qb
                return (a + b) % (1 << w)

            actual, expected = verify_circuit(circuit_builder, width)
            assert actual == expected, f"width={width}, a={a}, b={b}: expected {expected}, got {actual}"

    @pytest.mark.parametrize("width", [9, 16, 32])
    def test_dynamic_fallback_widths(self, verify_circuit, width):
        """Widths > 8 use dynamic generation and produce correct results."""
        # Test boundary values
        max_val = min((1 << width) - 1, 255)  # Cap for reasonable test time

        def circuit_builder(w=width, mv=max_val):
            qa = ql.qint(1, width=w)
            qb = ql.qint(mv, width=w)
            _r = qa + qb
            return (1 + mv) % (1 << w)

        actual, expected = verify_circuit(circuit_builder, width)
        assert actual == expected, f"width={width}: expected {expected}, got {actual}"


@pytest.mark.hardcoded_validation
class TestHardcodedSequenceStructure:
    """Verify hardcoded sequences have correct structure."""

    @pytest.mark.parametrize("width", [1, 2, 3, 4, 5, 6, 7, 8])
    def test_qq_add_circuit_depth(self, width):
        """QQ_add hardcoded produces expected circuit depth."""
        # Circuit depth should match formula: 5 * bits - 2
        expected_depth = 5 * width - 2

        ql.circuit()
        qa = ql.qint(0, width=width)
        qb = ql.qint(0, width=width)
        _r = qa + qb
        circ = ql.circuit()

        # Note: actual depth may differ due to optimization
        # This test documents the expected raw sequence structure
        assert circ.depth >= 1, f"Circuit should have non-zero depth for width {width}"

    @pytest.mark.parametrize("width", [1, 2, 3, 4, 5, 6, 7, 8])
    def test_cqq_add_circuit_executes(self, width):
        """cQQ_add hardcoded executes without error."""
        ql.circuit()
        qa = ql.qint(1, width=width)
        qb = ql.qint(1, width=width)

        # Controlled addition via conditional context
        ctrl = ql.qint(1, width=1)
        with ql.control(ctrl):
            _r = qa + qb

        circ = ql.circuit()
        assert circ.depth >= 1, f"Controlled circuit should have depth for width {width}"
