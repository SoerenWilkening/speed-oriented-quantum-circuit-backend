    ```python
    def __floordiv__(self, divisor):
        """Floor division: self // divisor

        Implements restoring division via repeated subtraction
        per arXiv:1809.09732 approach at Python level.

        Args:
            divisor: int or qint to divide by

        Returns:
            qint containing quotient

        Raises:
            ZeroDivisionError: if divisor is 0
            TypeError: if divisor is not int or qint

        Examples:
            >>> a = qint(10, width=8)
            >>> q = a // 3
            >>> q.width
            8
        """
        # Validate divisor
        if isinstance(divisor, int):
            if divisor == 0:
                raise ZeroDivisionError("Division by zero")
            if divisor < 0:
                raise NotImplementedError("Negative divisor not yet supported")

            return self._floordiv_classical(divisor)

        elif isinstance(divisor, qint):
            # Quantum divisor: more complex
            return self._floordiv_quantum(divisor)

        else:
            raise TypeError("Divisor must be int or qint")

    cdef _floordiv_classical(self, int divisor):
        """Floor division by classical (known) divisor."""

        # Special case: division by 1
        if divisor == 1:
            return self.copy() if hasattr(self, 'copy') else qint(self.value, width=self.bits)

        # Special case: power-of-2 division (just bit shift)
        if (divisor & (divisor - 1)) == 0 and divisor > 0:
            shift = divisor.bit_length() - 1
            # For power-of-2, quotient is right-shifted value
            # Quantum right-shift via qubit remapping
            # For now, use full algorithm; optimize in Phase 8
            pass

        # General case: restoring division algorithm
        # Start from MSB, try subtracting divisor * 2^bit
        quotient = qint(0, width=self.bits)

        # Work on copy of dividend (preserve self)
        remainder = qint(0, width=self.bits)
        remainder ^= self  # remainder = self via XOR from 0

        # Process bits from MSB to LSB
        for bit_pos in range(self.bits - 1, -1, -1):
            # Trial divisor for this bit position
            trial = divisor << bit_pos

            # Skip if trial > max possible value (optimization)
            if trial >= (1 << self.bits):
                continue

            # Quantum comparison: can we subtract?
            # cmp = (remainder >= trial) as qbool
            cmp = remainder >= trial

            # Conditional subtraction: if cmp, subtract trial from remainder
            with cmp:
                remainder -= trial
                # Set quotient bit (quotient[bit_pos] = 1)
                # Use XOR with 1 shifted to position
                # quotient.qubits[bit_pos] ^= 1 (conditional on cmp)
                # Implementation: quotient[bit_pos] should be set to cmp value
                # For now, use simpler approach: quotient += (1 << bit_pos)
                quotient += (1 << bit_pos)

        return quotient
    ```

    Note: This is the restoring division algorithm adapted for quantum.
    Each iteration uses comparison (O(n^2) gates) and conditional subtraction (O(n^2) gates),
    for total O(n * n^2) = O(n^3) gates.

    ```python
    cdef _floordiv_quantum(self, qint divisor):
        """Floor division by quantum divisor.

        More complex than classical case because divisor value is unknown.
        Uses repeated comparison and conditional subtraction.
        """
        # Determine width for operations
        comp_bits = max(self.bits, divisor.bits)

        # Allocate quotient and working remainder
        quotient = qint(0, width=comp_bits)
        remainder = qint(0, width=comp_bits)
        remainder ^= self  # Copy self to remainder

        # Simple repeated subtraction approach
        # While remainder >= divisor, subtract divisor and increment quotient
        # Maximum iterations = 2^bits (worst case)
        max_iterations = 1 << comp_bits

        # For practicality, limit iterations
        # Each iteration adds O(n^2) gates
        # Warn if width is large
        if comp_bits > 16:
            import warnings
            warnings.warn(
                f"Quantum division with {comp_bits}-bit integers generates "
                f"large circuits. Consider using classical divisor if possible.",
                UserWarning
            )

        # Bound iterations by quotient bit width
        for _ in range(comp_bits):
            # Shift approach: try subtracting divisor * 2^bit from MSB down
            pass

        # For now, use simpler iterative approach with bounded iterations
        # This is correct but potentially inefficient
        for iteration in range(min(max_iterations, 256)):  # Cap at 256 iterations
            # Compare: remainder >= divisor?
            cmp = remainder >= divisor

            # Conditional subtraction and increment
            with cmp:
                remainder -= divisor
                quotient += 1

            # Note: This produces a superposition of quotient values
            # Each iteration is applied to all branches

        return quotient
    ```

    ```python
    def __mod__(self, divisor):
        """Modulo operation: self % divisor

        Computes remainder after floor division.
        Uses: remainder = self - (self // divisor) * divisor

        Args:
            divisor: int or qint to divide by

        Returns:
            qint containing remainder

        Raises:
            ZeroDivisionError: if divisor is 0
            TypeError: if divisor is not int or qint

        Examples:
            >>> a = qint(10, width=8)
            >>> r = a % 3
            >>> r.width
            8
        """
        if isinstance(divisor, int):
            if divisor == 0:
                raise ZeroDivisionError("Modulo by zero")
        elif isinstance(divisor, qint):
            # Cannot check for zero at compile time for quantum divisor
            pass
        else:
            raise TypeError("Divisor must be int or qint")

        # Compute quotient
        quotient = self // divisor

        # Compute remainder = self - quotient * divisor
        product = quotient * divisor
        remainder = self - product

        return remainder
    ```

    ```python
    def __mod__(self, divisor):

                raise ZeroDivisionError("Modulo by zero")
            return self._mod_classical(divisor)
        elif isinstance(divisor, qint):
            return self._mod_quantum(divisor)
        else:
            raise TypeError("Divisor must be int or qint")

    cdef _mod_classical(self, int divisor):
        """Modulo by classical divisor - more efficient than full division."""
        # Copy self to remainder
        remainder = qint(0, width=self.bits)
        remainder ^= self

        # Repeated conditional subtraction until remainder < divisor
        for bit_pos in range(self.bits - 1, -1, -1):
            trial = divisor << bit_pos
            if trial >= (1 << self.bits):
                continue

            cmp = remainder >= trial
            with cmp:
                remainder -= trial

        return remainder
    ```

    ```python
    def __divmod__(self, divisor):
        """Return (quotient, remainder) tuple.

        More efficient than calling // and % separately as division
        is only performed once.

        Args:
            divisor: int or qint to divide by

        Returns:
            tuple (quotient: qint, remainder: qint)

        Raises:
            ZeroDivisionError: if divisor is 0

        Examples:
            >>> a = qint(10, width=8)
            >>> q, r = divmod(a, 3)
            >>> q.width, r.width
            (8, 8)
        """
        if isinstance(divisor, int):
            if divisor == 0:
                raise ZeroDivisionError("Division by zero")
            return self._divmod_classical(divisor)
        elif isinstance(divisor, qint):
            return self._divmod_quantum(divisor)
        else:
            raise TypeError("Divisor must be int or qint")

    cdef _divmod_classical(self, int divisor):
        """Efficient divmod for classical divisor."""
        quotient = qint(0, width=self.bits)
        remainder = qint(0, width=self.bits)
        remainder ^= self  # Copy self to remainder

        # Restoring division - compute both at once
        for bit_pos in range(self.bits - 1, -1, -1):
            trial = divisor << bit_pos
            if trial >= (1 << self.bits):
                continue

            cmp = remainder >= trial
            with cmp:
                remainder -= trial
                quotient += (1 << bit_pos)

        return (quotient, remainder)

    cdef _divmod_quantum(self, qint divisor):
        """Divmod for quantum divisor."""
        comp_bits = max(self.bits, divisor.bits)

        quotient = qint(0, width=comp_bits)
        remainder = qint(0, width=comp_bits)
        remainder ^= self

        # Same iterative approach as _floordiv_quantum
        for bit_pos in range(comp_bits - 1, -1, -1):
            # For quantum divisor, we can't pre-compute trial = divisor << bit_pos
            # Need quantum shift or different algorithm
            # For now, use simple iterative subtraction
            pass

        # Simpler iterative approach
        for _ in range(min(1 << comp_bits, 256)):
            cmp = remainder >= divisor
            with cmp:
                remainder -= divisor
                quotient += 1

        return (quotient, remainder)
    ```

    ```python
    def __rfloordiv__(self, other):
        """Reverse floor division: other // self (for int // qint).

        Args:
            other: int to divide

        Returns:
            qint containing quotient
        """
        if isinstance(other, int):
            # Convert to qint and divide
            other_qint = qint(other, width=self.bits)
            return other_qint // self
        else:
            return NotImplemented

    def __rmod__(self, other):
        """Reverse modulo: other % self (for int % qint).

        Args:
            other: int to take modulo of

        Returns:
            qint containing remainder
        """
        if isinstance(other, int):
            other_qint = qint(other, width=self.bits)
            return other_qint % self
        else:
            return NotImplemented

    def __rdivmod__(self, other):
        """Reverse divmod: divmod(other, self) for int, qint.

        Args:
            other: int to divide

        Returns:
            tuple (quotient, remainder)
        """
        if isinstance(other, int):
            other_qint = qint(other, width=self.bits)
            return divmod(other_qint, self)
        else:
            return NotImplemented
    ```
