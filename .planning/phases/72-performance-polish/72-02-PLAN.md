2. Modify `toffoli_QQ_add(int bits)` to check hardcoded sequences before dynamic generation. Insert BEFORE the existing cache check, following the exact pattern from `IntegerAddition.c` lines 160-170:

```c
// Use hardcoded sequences for widths 1-8
if (bits <= TOFFOLI_HARDCODED_MAX_WIDTH) {
    const sequence_t *hardcoded = get_hardcoded_toffoli_QQ_add(bits);
    if (hardcoded != NULL) {
        // SAFETY: Const cast is safe -- static sequences have program lifetime
        precompiled_toffoli_QQ_add[bits] = (sequence_t *)hardcoded;
        return (sequence_t *)hardcoded;
    }
}
```

Place this right after the bounds check and BEFORE the existing cache check `if (precompiled_toffoli_QQ_add[bits] != NULL)`. Actually, place it AFTER the cache check -- the cache check is O(1) and the hardcoded lookup might be slightly slower. On first call for a given width, the cache is empty, so it falls through to the hardcoded check. On subsequent calls, the cache returns immediately.

Better approach: Place the hardcoded check BETWEEN the cache check and the dynamic generation. This way:
- First call: cache miss -> hardcoded hit -> store in cache -> return
- Subsequent calls: cache hit -> return immediately

3. Modify `toffoli_cQQ_add(int bits)` with the same pattern:

```c
// Use hardcoded sequences for widths 1-8
if (bits <= TOFFOLI_HARDCODED_MAX_WIDTH) {
    const sequence_t *hardcoded = get_hardcoded_toffoli_cQQ_add(bits);
    if (hardcoded != NULL) {
        precompiled_toffoli_cQQ_add[bits] = (sequence_t *)hardcoded;
        return (sequence_t *)hardcoded;
    }
}
```

Place between the cache check and the dynamic generation, same as QQ.

**CRITICAL ownership note:** Hardcoded sequences are static const (program lifetime). The precompiled cache stores them without ownership. `toffoli_sequence_free()` must NEVER be called on these. The existing ownership contract already handles this: QQ and cQQ sequences are cached and never freed. No changes needed to `toffoli_sequence_free()`.

```python
"""Tests for hardcoded Toffoli sequences and T-count reporting (Phase 72)."""
import pytest
import quantum_language as ql
from quantum_language import qint, circuit

# ============================================================================
# T-count reporting tests (INF-04)
# ============================================================================

class TestTCount:
    """Verify T-count appears in gate_counts and equals 7 * CCX."""

    def test_t_count_in_gate_counts(self):
        """gate_counts dict has 'T' key."""
        c = circuit()
        a = qint(4)
        b = qint(4)
        a += b
        counts = c.gate_counts
        assert 'T' in counts, "gate_counts missing 'T' key"

    def test_t_count_equals_7_times_ccx(self):
        """T = 7 * CCX for Toffoli arithmetic."""
        c = circuit()
        a = qint(4)
        b = qint(4)
        a += b
        counts = c.gate_counts
        assert counts['T'] == 7 * counts['CCX'], (
            f"T={counts['T']} != 7*CCX={7*counts['CCX']}"
        )

    def test_t_count_zero_for_qft(self):
        """T = 0 for QFT arithmetic (no CCX gates)."""
        c = circuit()
        ql.option('fault_tolerant', False)
        a = qint(4)
        b = qint(4)
        a += b
        counts = c.gate_counts
        assert counts['CCX'] == 0, "QFT should have no CCX gates"
        assert counts['T'] == 0, "QFT should have T=0"
        ql.option('fault_tolerant', True)  # restore default

    def test_t_count_scales_with_width(self):
        """Wider additions have more CCX -> more T gates."""
        t_counts = []
        for width in [2, 4, 8]:
            c = circuit()
            a = qint(width)
            b = qint(width)
            a += b
            t_counts.append(c.gate_counts['T'])
        assert t_counts[0] < t_counts[1] < t_counts[2], (
            f"T-count should increase with width: {t_counts}"
        )

# ============================================================================
# Hardcoded Toffoli sequence tests (INF-03)
# ============================================================================

class TestHardcodedToffoliSequences:
    """Verify hardcoded sequences produce correct results."""

    @pytest.mark.parametrize("width", [1, 2, 3, 4, 5, 6, 7, 8])
    def test_qq_add_hardcoded_correctness(self, width):
        """QQ addition with hardcoded sequences matches expected results."""
        # Test a few input pairs per width
        max_val = (1 << width) - 1
        test_pairs = [(0, 0), (1, 0), (0, 1), (1, 1)]
        if width >= 2:
            test_pairs.extend([(max_val, 1), (max_val, max_val)])

        for a_val, b_val in test_pairs:
            c = circuit()
            a = qint(width, a_val)
            b = qint(width, b_val)
            a += b
            expected = (a_val + b_val) % (1 << width)
            result = a.measure()
            assert result == expected, (
                f"Width {width}: {a_val} + {b_val} = {result}, expected {expected}"
            )

    @pytest.mark.parametrize("width", [1, 2, 3, 4])
    def test_qq_add_gate_purity(self, width):
        """Hardcoded QQ Toffoli addition uses only X/CX/CCX gates."""
        c = circuit()
        a = qint(width)
        b = qint(width, 1)
        a += b
        counts = c.gate_counts
        assert counts['H'] == 0, f"Width {width}: H gates found in Toffoli add"
        assert counts['P'] == 0, f"Width {width}: P gates found in Toffoli add"

    @pytest.mark.parametrize("width", [2, 4, 8])
    def test_cqq_add_hardcoded_correctness(self, width):
        """Controlled QQ addition with hardcoded sequences works correctly."""
        c = circuit()
        ctrl = qint(1, 1)  # control = |1>
        a = qint(width, 3)
        b = qint(width, 2)
        with ctrl:
            a += b
        expected = (3 + 2) % (1 << width)
        result = a.measure()
        assert result == expected, (
            f"Controlled add width {width}: 3 + 2 = {result}, expected {expected}"
        )

    def test_hardcoded_no_regression_existing_tests(self):
        """Sanity check: basic Toffoli add still works after hardcoded integration."""
        c = circuit()
        a = qint(4, 7)
        b = qint(4, 5)
        a += b
        assert a.measure() == 12, "4-bit: 7 + 5 should be 12"

    @pytest.mark.parametrize("width", [2, 4])
    def test_subtraction_still_works(self, width):
        """Subtraction (inverted adder) works with hardcoded sequences."""
        c = circuit()
        a = qint(width, 5 % (1 << width))
        b = qint(width, 3 % (1 << width))
        a -= b
        expected = (5 - 3) % (1 << width)
        result = a.measure()
        assert result == expected, (
            f"Width {width}: 5 - 3 = {result}, expected {expected}"
        )
```
