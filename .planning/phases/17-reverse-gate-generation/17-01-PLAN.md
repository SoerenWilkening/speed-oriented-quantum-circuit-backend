```cython
def reverse_instruction_range(int start_layer, int end_layer):
    """Reverse gates in circuit from start_layer to end_layer (exclusive).

    Parameters
    ----------
    start_layer : int
        Starting layer index (inclusive)
    end_layer : int
        Ending layer index (exclusive)

    Notes
    -----
    Reverses gates in LIFO order, appending adjoint gates to circuit.
    Phase gates have their angles negated. Self-adjoint gates (X, H, CX)
    are their own inverses.
    """
    global _circuit
    if not _circuit_initialized:
        raise RuntimeError("Circuit not initialized")
    reverse_circuit_range(_circuit, start_layer, end_layer)
```

3. Add helper function to get current layer count:
```cython
def get_current_layer():
    """Get current layer count in circuit.

    Returns
    -------
    int
        Number of used layers in circuit.
    """
    global _circuit
    if not _circuit_initialized:
        return 0
    return (<circuit_t*>_circuit).used_layer
```

```python
def test_reverse_self_adjoint_gates():
    """Test that self-adjoint gates (X, H) reverse correctly."""
    ql.circuit()

    # Record start layer
    start = ql.get_current_layer()

    # Create qbool and apply NOT (X gate)
    a = ql.qbool(False)
    ~a  # Apply X gate

    # Record end layer
    end = ql.get_current_layer()

    # Get gate count before reversal
    before_gates = ql.circuit().gate_count

    # Reverse the instruction range
    ql.reverse_instruction_range(start, end)

    # After reversal, should have double the gates (original + inverted)
    # X reversed is X, so circuit should have X...X pattern
    after_gates = ql.circuit().gate_count

    # Gate count should increase (reversed gates appended)
    assert after_gates > before_gates, f"Expected more gates after reversal, got {before_gates} -> {after_gates}"
    print("PASS: test_reverse_self_adjoint_gates")

def test_reverse_phase_gates():
    """Test that phase gates invert correctly (P(t) -> P(-t))."""
    ql.circuit()

    start = ql.get_current_layer()

    # Create qints and perform addition (uses phase gates via QFT)
    a = ql.qint(3, width=4)
    a += 1  # Addition uses QFT which has phase gates

    end = ql.get_current_layer()
    before_gates = ql.circuit().gate_count

    # Reverse
    ql.reverse_instruction_range(start, end)

    after_gates = ql.circuit().gate_count
    assert after_gates > before_gates, f"Expected more gates after reversal"
    print("PASS: test_reverse_phase_gates")

def test_reverse_empty_range():
    """Test that empty range (start == end) is no-op."""
    ql.circuit()

    a = ql.qint(5, width=4)
    before_gates = ql.circuit().gate_count
    before_layers = ql.get_current_layer()

    # Reverse empty range
    ql.reverse_instruction_range(0, 0)

    after_gates = ql.circuit().gate_count
    after_layers = ql.get_current_layer()

    assert before_gates == after_gates, "Empty range should not add gates"
    assert before_layers == after_layers, "Empty range should not add layers"
    print("PASS: test_reverse_empty_range")

def test_reverse_controlled_gates():
    """Test that controlled gates preserve control structure."""

    end = ql.get_current_layer()

    # Reverse
    ql.reverse_instruction_range(start, end)

    after_gates = ql.circuit().gate_count
    assert after_gates > before_gates, f"Expected more gates after reversal"
    print("PASS: test_reverse_controlled_gates")
```
