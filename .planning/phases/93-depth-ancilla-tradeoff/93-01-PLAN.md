```python
    global _tradeoff_policy, _arithmetic_ops_performed

        return _tradeoff_policy
    if value not in ('auto', 'min_depth', 'min_qubits'):
        raise ValueError(
            f"Invalid tradeoff value: {value!r}. "

        )
    if _arithmetic_ops_performed:
        raise RuntimeError(
            "Cannot change tradeoff policy after arithmetic operations have been "
            "performed. Set ql.option('tradeoff', ...) before any +, -, * operations."
        )
    _tradeoff_policy = value
    _validate_circuit()
    cdef circuit_s *circ = <circuit_s*><circuit_t*><unsigned long long>_get_circuit()
    if value == 'min_qubits':
        circ.cla_override = 1          # Force RCA everywhere
        circ.tradeoff_min_depth = 0
    elif value == 'min_depth':
        circ.cla_override = 0          # Allow CLA dispatch
        circ.tradeoff_auto_threshold = 2  # CLA for all widths >= 2
        circ.tradeoff_min_depth = 1    # Enable CLA subtraction (Plan 02)
    else:  # 'auto'
        circ.cla_override = 0          # Allow CLA dispatch
        circ.tradeoff_auto_threshold = 4  # Empirical threshold (may be tuned)
        circ.tradeoff_min_depth = 0
```
