**Implement `circuit_to_openqasm(circuit_t *circ, const char *path)`:**

This function reuses `circuit_to_qasm_string()` from Plan 01 to avoid duplicating gate export logic:

```c
int circuit_to_openqasm(circuit_t *circ, const char *path) {
    if (circ == NULL || path == NULL) {
        return -1;
    }

    // Get QASM string using the string export function
    char *qasm = circuit_to_qasm_string(circ);
    if (qasm == NULL) {
        return -1;
    }

    // Build file path
    char filepath[512];
    snprintf(filepath, sizeof(filepath), "%s/circuit.qasm", path);

    // Write to file
    FILE *f = fopen(filepath, "w");
    if (f == NULL) {
        free(qasm);
        return -1;
    }

    fputs(qasm, f);
    fclose(f);
    free(qasm);
    return 0;
}
```
