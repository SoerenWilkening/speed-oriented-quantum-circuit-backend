2. **__repr__ method:**
   ```python
   def __repr__(self):
       """Compact format: ql.array<qint:8, shape=(3,)>[1, 2, 3]

       Truncates large arrays with ellipsis (...) NumPy-style.
       """
       dtype_name = self._dtype.__name__  # "qint" or "qbool"

       # Format type info
       if dtype_name == "qint":
           type_info = f"<qint:{self._width}, shape={self._shape}>"
       else:
           type_info = f"<qbool, shape={self._shape}>"

       # Format elements
       if len(self._shape) == 1:
           elem_str = self._format_1d()
       else:
           elem_str = self._format_nested(self._elements, self._shape)

       return f"ql.array{type_info}[{elem_str}]"
   ```

3. **_format_1d helper (with truncation):**
   ```python
   def _format_1d(self):
       """Format 1D array with truncation."""
       threshold = 6  # Show first 3 and last 3 if > threshold

       if len(self._elements) <= threshold:
           return ", ".join(str(e.value) for e in self._elements)
       else:
           first = ", ".join(str(e.value) for e in self._elements[:3])
           last = ", ".join(str(e.value) for e in self._elements[-3:])
           return f"{first}, ..., {last}"
   ```

4. **_format_nested helper (for multi-dimensional):**
   ```python
   def _format_nested(self, elements, shape):
       """Format multi-dimensional array with nested brackets."""
       if len(shape) == 1:
           # Base case: format this dimension
           if len(elements) <= 6:
               return ", ".join(str(e.value) for e in elements)
           else:
               first = ", ".join(str(e.value) for e in elements[:3])
               last = ", ".join(str(e.value) for e in elements[-3:])
               return f"{first}, ..., {last}"

       # Recursive case
       stride = 1
       for s in shape[1:]:
           stride *= s

       chunks = []
       for i in range(0, len(elements), stride):
           chunk = elements[i:i+stride]
           chunks.append("[" + self._format_nested(chunk, shape[1:]) + "]")

       return ", ".join(chunks)
   ```

5. **__str__ same as __repr__:**
   ```python
   def __str__(self):
       return self.__repr__()
   ```
