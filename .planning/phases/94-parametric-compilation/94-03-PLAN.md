"""Tests for parametric compilation (Phase 94).

Verifies:
- PAR-01: @ql.compile(parametric=True) API
- PAR-02: Parametric replay correctness via Qiskit simulation
- PAR-03: Toffoli CQ structural fallback
- PAR-04: Oracle decorator forces non-parametric
- FIX-04: Mode flag cache invalidation
"""

import warnings

import pytest
from qiskit import QuantumCircuit, transpile
from qiskit_aer import AerSimulator

import quantum_language as ql
from quantum_language import qint

warnings.filterwarnings("ignore", message="Value .* exceeds")

SIM = AerSimulator(method="statevector", max_parallel_threads=4)


def simulate_and_measure(width):
    """Export current circuit to Qiskit and measure all qubits."""
    qasm_str = ql.qasm()
    qc = QuantumCircuit.from_qasm_str(qasm_str)
    qc.measure_all()
    qc_t = transpile(qc, SIM)
    result = SIM.run(qc_t, shots=1).result()
    counts = result.get_counts()
    bitstring = list(counts.keys())[0]
    return bitstring
```

class TestParametricQFTCorrectness:
    """Verify parametric replay correctness in QFT mode (rotation-based).

    QFT arithmetic uses rotations where classical values affect only
    gate angles, not topology. Parametric mode should detect this and
    produce correct results for multiple distinct classical values.
    """

    def test_parametric_add_two_values(self):
        """Parametric add produces correct results for two different classical values."""
        @ql.compile(parametric=True)
        def add_val(x, val):
            x += val
            return x

        # Value 1: 3 + 2 = 5
        c = ql.circuit()
        a = ql.qint(3, width=4)
        result = add_val(a, 2)
        qasm1 = ql.qasm()

        # Value 2: 3 + 5 = 8
        c = ql.circuit()
        a = ql.qint(3, width=4)
        result = add_val(a, 5)
        qasm2 = ql.qasm()

        # Both should produce valid circuits (different QASM due to different rotations)
        assert len(qasm1) > 0
        assert len(qasm2) > 0

    def test_parametric_add_simulation_value1(self):
        """Simulate parametric add with classical value 2: 3 + 2 = 5."""
        @ql.compile(parametric=True)
        def add_val(x, val):
            x += val
            return x

        c = ql.circuit()
        a = ql.qint(3, width=4)
        result = add_val(a, 2)
        bits = simulate_and_measure(4)
        measured = int(bits[:4], 2)  # First 4 bits
        assert measured == 5, f"Expected 5, got {measured}"

    def test_parametric_add_simulation_value2(self):
        """Simulate parametric add with classical value 5: 3 + 5 = 8."""
        @ql.compile(parametric=True)
        def add_val(x, val):
            x += val
            return x

        c = ql.circuit()
        a = ql.qint(3, width=4)
        result = add_val(a, 5)
        bits = simulate_and_measure(4)
        measured = int(bits[:4], 2)
        assert measured == 8, f"Expected 8, got {measured}"

    def test_parametric_add_multiple_values(self):
        """Parametric add correct for values 1 through 6."""
        @ql.compile(parametric=True)
        def add_val(x, val):
            x += val
            return x

        for val in range(1, 7):
            c = ql.circuit()
            a = ql.qint(0, width=4)
            result = add_val(a, val)
            bits = simulate_and_measure(4)
            measured = int(bits[:4], 2)
            assert measured == val, f"0 + {val}: expected {val}, got {measured}"

    Parametric mode should detect structural parameters and fall back
    to per-value caching automatically.
    """

    def test_toffoli_cq_detected_as_structural(self):
        """Toffoli CQ add detected as structural (different topology per value)."""
        @ql.compile(parametric=True)
        def add_val(x, val):
            x += val
            return x

        # First call
        c = ql.circuit()
        ql.option('fault_tolerant', True)
        a = ql.qint(0, width=4)
        _ = add_val(a, 3)

        # Second call with different value
        c = ql.circuit()
        ql.option('fault_tolerant', True)
        a = ql.qint(0, width=4)
        _ = add_val(a, 5)

        assert add_val._parametric_safe is False, \

class TestParametricEdgeCases:
    """Edge cases for parametric compilation."""

    def test_no_classical_args_is_noop(self):
        """parametric=True with no classical args behaves normally."""
        c = ql.circuit()

        @ql.compile(parametric=True)
        def inc(x):
            x += 1
            return x

        a = ql.qint(3, width=4)
        result = inc(a)
        # Should work — no classical args means parametric is no-op
        assert result is not None

    def test_clear_cache_resets_parametric_state(self):
        """clear_cache() resets parametric probe state."""
        @ql.compile(parametric=True)
        def add_val(x, val):
            x += val
            return x

        # Trigger probe
        c = ql.circuit()
        a = ql.qint(0, width=4)
        _ = add_val(a, 3)
        assert add_val._parametric_probed is True

        # Clear cache
        add_val.clear_cache()
        assert add_val._parametric_probed is False
        assert add_val._parametric_safe is None

    def test_is_parametric_property(self):
        """is_parametric property returns correct value."""
        @ql.compile(parametric=True)
        def f(x, val):
            x += val
            return x
        assert f.is_parametric is True

        @ql.compile
        def g(x, val):
            x += val
            return x
        assert g.is_parametric is False

    def test_same_classical_value_is_cache_hit(self):
        """Same classical args should be a cache hit (not re-probe)."""
        @ql.compile(parametric=True, debug=True)
        def add_val(x, val):
            x += val
            return x

        c = ql.circuit()
        a = ql.qint(0, width=4)
        _ = add_val(a, 3)  # First call (capture)

        a2 = ql.qint(1, width=4)
        _ = add_val(a2, 3)  # Same classical value — should hit cache
        if add_val.stats is not None:
            assert add_val.stats['cache_hit'] is True
```
