parametric : bool
    If True, enable parametric compilation mode.  The function is
    captured once and replayed with different classical argument
    values without re-capturing the gate sequence.  Falls back to
    per-value caching when classical arguments affect gate topology

def _lambda_cache_key(predicate, register_widths):
    """Build cache key for lambda/predicate oracles including closure variable values.
    ...
    Returns
    -------
    tuple
        ``(source_hash, closure_values, arithmetic_mode_int, cla_override, tradeoff_policy, register_widths_tuple)``
    """
    src_hash = _compute_source_hash(predicate)

    # Extract closure variable values (only hashable primitive types)
    closure_values = ()
    if hasattr(predicate, "__closure__") and predicate.__closure__:
        vals = []
        for cell in predicate.__closure__:
            try:
                contents = cell.cell_contents
            except ValueError:
                continue
            if isinstance(contents, int | float | str | bool):
                vals.append(contents)
        closure_values = tuple(vals)

    arithmetic_mode = 1 if option("fault_tolerant") else 0
    cla_override = 0 if option("cla") else 1
    tradeoff_policy = option("tradeoff")
    return (src_hash, closure_values, arithmetic_mode, cla_override, tradeoff_policy, tuple(register_widths))
```

class TestModeFlagCacheKey:
    """Verify that switching arithmetic modes invalidates compile cache (FIX-04)."""

    def test_arithmetic_mode_switch_invalidates_cache(self):
        """Switching fault_tolerant between calls should not reuse stale cache."""
        c = ql.circuit()
        @ql.compile
        def add_vals(x, y):
            return x + y

        # Call in QFT mode
        ql.option('fault_tolerant', False)
        a = ql.qint(3, width=4)
        b = ql.qint(2, width=4)
        _ = add_vals(a, b)
        cache_size_after_qft = len(add_vals._cache)

        # Switch to Toffoli mode and call again with same widths
        c = ql.circuit()
        ql.option('fault_tolerant', True)
        a = ql.qint(3, width=4)
        b = ql.qint(2, width=4)
        _ = add_vals(a, b)

        # Should have separate cache entries (not reuse QFT gates for Toffoli)
        # After new circuit, cache is cleared — so we check it captured fresh
        assert len(add_vals._cache) >= 1

    def test_tradeoff_switch_invalidates_cache(self):
        """Switching tradeoff policy between calls should cause re-capture."""
        c = ql.circuit()
        ql.option('fault_tolerant', True)

        @ql.compile
        def add_vals(x, y):
            return x + y

        # Call in auto mode
        a = ql.qint(3, width=6)
        b = ql.qint(2, width=6)
        _ = add_vals(a, b)
        auto_cache_size = len(add_vals._cache)

        # New circuit, min_depth mode
        c = ql.circuit()
        ql.option('fault_tolerant', True)
        ql.option('tradeoff', 'min_depth')
        a = ql.qint(3, width=6)
        b = ql.qint(2, width=6)
        _ = add_vals(a, b)

        # Cache cleared on new circuit, but should still capture fresh
        assert len(add_vals._cache) >= 1

    def test_same_mode_reuses_cache(self):
        """Same mode flags should produce cache hit on second call."""
        c = ql.circuit()
        ql.option('fault_tolerant', True)

        @ql.compile(debug=True)
        def add_vals(x, y):
            return x + y

        a = ql.qint(3, width=4)
        b = ql.qint(2, width=4)
        _ = add_vals(a, b)  # Capture

        a2 = ql.qint(5, width=4)
        b2 = ql.qint(1, width=4)
        _ = add_vals(a2, b2)  # Should be cache hit (same mode, same widths, same classical args=none)

        assert add_vals.stats['cache_hit'] == True
```
