  2. Add cdef extern declaration to `src/quantum_language/_core.pxd`:
     ```cython
     cdef extern from "hot_path_1.h":
         void hot_path_<name>(
             circuit_t *circ,
             const unsigned int *self_qubits,
             int self_bits,
             const unsigned int *other_qubits,
             int other_bits,
             int64_t classical_value,
             int invert,

             const unsigned int *ancilla,
             int num_ancilla

     ```

  3. Replace the Cython implementation in the relevant .pxi file:
     - Keep the Python-facing method signature identical
     - Extract all data from Python objects BEFORE the C call
     - Build C arrays on the Cython stack (cdef unsigned int self_qa[64], etc.)
     - Fetch circuit_t*, controlled flag, control_qubit from Python globals
     - Call the C function with extracted data
     - Apply CYT-04 (nogil): wrap the C function call in `with nogil:` block since the C function makes no Python callbacks

     Example replacement pattern for addition_inplace:
     ```cython
     cdef addition_inplace(self, other, int invert=False):
         cdef circuit_t *_circuit = <circuit_t*><unsigned long long>_get_circuit()
         cdef bint _controlled = _get_controlled()
         cdef object _control_bool = _get_control_bool()
         cdef unsigned int self_qa[64]
         cdef unsigned int other_qa[64]
         cdef unsigned int ancilla_qa[16]
         cdef int self_offset = 64 - self.bits
         cdef int i, other_bits = 0
         cdef int64_t classical_value = 0
         cdef unsigned int control_qubit = 0
         cdef int num_ancilla = NUMANCILLY
         cdef unsigned int[:] ancilla_arr
         cdef unsigned int[:] control_qubits

         for i in range(self.bits):
             self_qa[i] = self.qubits[self_offset + i]

         if type(other) == int:
             classical_value = <int64_t>other
         elif isinstance(other, qint):
             other_bits = (<qint>other).bits
             cdef int other_offset = 64 - other_bits
             for i in range(other_bits):
                 other_qa[i] = (<qint>other).qubits[other_offset + i]
         else:
             raise ValueError()

         if _controlled:
             control_qubits = (<qint>_control_bool).qubits
             control_qubit = control_qubits[63]

         ancilla_arr = _get_ancilla()
         for i in range(num_ancilla):
             ancilla_qa[i] = ancilla_arr[i]

         with nogil:
             hot_path_<name>(_circuit, self_qa, self.bits,
                            other_qa if other_bits > 0 else NULL,
                            other_bits, classical_value, invert,
                            _controlled, control_qubit, ancilla_qa, num_ancilla)

         return self
     ```
