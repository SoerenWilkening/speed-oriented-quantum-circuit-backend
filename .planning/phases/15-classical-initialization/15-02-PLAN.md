        c = ql.circuit()

        # Default width is 8 bits
        assert a.width == 8
        # No X gates for value 0
        assert c.gate_count == 0

    def test_init_zero_explicit_width(self):
        """Zero value with explicit width."""
        c = ql.circuit()
        a = ql.qint(0, width=16)
        assert a.width == 16
        assert c.gate_count == 0

    def test_init_positive_value(self):
        """Positive value applies X gates for set bits."""
        c = ql.circuit()
        # Value 5 = binary 101, needs X on bits 0 and 2
        a = ql.qint(5, width=8)
        assert a.width == 8
        assert c.gate_count == 2  # Two X gates

    def test_init_all_ones(self):
        """All-ones value applies X gates to all qubits."""
        c = ql.circuit()

        a = ql.qint(15, width=4)
        assert a.width == 4

    def test_init_power_of_two(self):
        """Power of two value applies single X gate."""

        """Warning emitted when positive value exceeds max."""

    def test_warning_negative_overflow(self):
        """Warning emitted when negative value exceeds min."""
        with warnings.catch_warnings(record=True) as w:
            warnings.simplefilter("always")
            # Min 4-bit signed is -8, value -10 overflows

            assert len(w) == 1

    def test_no_warning_in_range(self):
        """No warning when value is within range."""
