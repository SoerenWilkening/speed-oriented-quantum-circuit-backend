**Add _variable_diffusion method to QWalkTree (place before local_diffusion):**

```python
def _variable_diffusion(self, depth):
    """Apply variable-branching local diffusion D_x at the given depth.

    Evaluates the predicate on each potential child to determine which
    are valid, then applies angle-conditional rotations based on the
    count of valid children d(x). When d(x) = 0 (all rejected), the
    diffusion is skipped entirely.

    Parameters
    ----------
    depth : int
        Depth level (1..max_depth). Must not be called for depth=0 (leaf).
    """
    import itertools
    from .qbool import qbool

    level_idx = self.max_depth - depth
    d_max = self.branching[level_idx]
    branch_reg = self.branch_registers[level_idx]
    w = branch_reg.width
    h_qubit_idx = self._height_qubit(depth)
    h_control = _make_qbool_wrapper(h_qubit_idx)
    is_root = (depth == self.max_depth)

    # For variable branching, the next level down is level_idx + 1
    # (children are one depth level below the current node)
    next_level_idx = level_idx + 1
    if next_level_idx >= self.max_depth:
        # This node's children are at the leaf level -- no branch register
        # In one-hot encoding, children at depth-1 may still have a branch register
        pass

    # --- Step 1: Allocate validity ancillae ---
    # One qbool per potential child. validity[i] = |1> means child i is VALID
    # (not rejected by predicate). Initially |0>, will be set to |1> for valid children.
    validity = [qbool() for _ in range(d_max)]

    # --- Step 2: Evaluate predicate on each child ---
    # For each child i: navigate to child state, evaluate predicate,
    # if NOT rejected -> set validity[i] to |1>, then undo navigation.
    child_branch_reg = self.branch_registers[level_idx] if level_idx < len(self.branch_registers) else None

    for i in range(d_max):
        # 2a. Navigate to child i: encode child index in the branch register
        # at the CURRENT level (the child's branch value within this node's children).
        # The branch register at level_idx stores which child was chosen at depth `depth`.
        # Setting it to i means "we're looking at child i".
        br = self.branch_registers[level_idx]
        bw = br.width
        for bit in range(bw):
            if (i >> (bw - 1 - bit)) & 1:
                emit_x(int(br.qubits[64 - bw + bit]))

        # 2b. Flip height register: move from depth to depth-1
        emit_x(self._height_qubit(depth))
        emit_x(self._height_qubit(depth - 1))

        # 2c. Evaluate predicate
        node = self.node
        accept_q, reject_q = self._predicate(node)

        # 2d. Store validity: valid = NOT rejected
        # CNOT from reject_q to validity[i], then X on validity[i]
        # This gives validity[i] = |1> when reject = |0> (valid child)
        reject_qubit = int(reject_q.qubits[63])
        validity_qubit = int(validity[i].qubits[63])
        reject_ctrl = _make_qbool_wrapper(reject_qubit)
        with reject_ctrl:
            emit_x(validity_qubit)
        emit_x(validity_qubit)  # Flip: now validity=|1> means valid

        # 2e. Uncompute predicate (reverse the predicate computation)
        if self._predicate_is_compiled:
            self._predicate.adjoint(node)
        # For non-compiled predicates, uncomputation is user's responsibility
        # but we still need to handle the returned qbools

        # 2f. Undo height flip
        emit_x(self._height_qubit(depth - 1))
        emit_x(self._height_qubit(depth))

        # 2g. Undo branch register encoding
        for bit in range(bw):
            if (i >> (bw - 1 - bit)) & 1:
                emit_x(int(br.qubits[64 - bw + bit]))

    # --- Step 3: Conditional angle dispatch ---
    # For each possible d value (1..d_max), emit rotation blocks
    # conditioned on "exactly d validity ancillae are |1>".
    # Each bit pattern with exactly d ones gets its own conditional block.

    # First: U_dagger (inverse state prep) conditioned on d value
    # Then: S_0 reflection
    # Then: U (forward state prep) conditioned on d value
    # Structure: U_inv * S_0 * U_fwd, all conditional on d(x)

    # --- Step 3a: U_dagger conditional on d(x) ---
    for d_val in range(1, d_max + 1):
        angle_data = self._variable_angles.get(d_val)
        if angle_data is None:
            continue
        phi_d = angle_data['phi']
        if is_root:
            phi_d = self._variable_root_angles.get(d_val, phi_d)
        cascade_ops_d = angle_data['cascade_ops']

        # Get all bit patterns with exactly d_val ones among d_max positions
        for pattern in itertools.combinations(range(d_max), d_val):
            # pattern = tuple of indices where validity[i] = |1>
            # Flip the zeros to match: X on validity qubits NOT in pattern
            zeros = [j for j in range(d_max) if j not in pattern]
            for z in zeros:
                emit_x(int(validity[z].qubits[63]))

            # Now all validity ancillae are |1> if this pattern matches.
            # Multi-controlled operations conditioned on ALL validity ancillae.
            # We need to control on h[depth] AND all validity ancillae.

            # Build control qubit list: h_qubit + all validity qubits
            ctrl_qubits = [h_qubit_idx] + [int(validity[j].qubits[63]) for j in range(d_max)]

            # U_dagger: inverse cascade then inverse parent-child split
            if d_val > 1 and cascade_ops_d:
                # Emit inverse cascade controlled on ALL ctrl_qubits
                # Use the innermost validity qubit as direct control,
                # and the rest via MCZ decomposition patterns
                _emit_cascade_multi_controlled(
                    branch_reg, cascade_ops_d, ctrl_qubits, sign=-1
                )
            # Inverse parent-child Ry
            _emit_multi_controlled_ry(
                self._height_qubit(depth - 1), -phi_d, ctrl_qubits
            )

            # Undo X flips
            for z in zeros:
                emit_x(int(validity[z].qubits[63]))

    # --- Step 3b: S_0 reflection ---
    # S_0 operates on the full local subspace: h[depth-1] + branch_reg
    # Controlled on h[depth]. This is independent of d(x).
    from ._gates import emit_mcz, emit_z
    from .diffusion import _collect_qubits

    h_child = _make_qbool_wrapper(self._height_qubit(depth - 1))
    s0_qubits = _collect_qubits(h_child, branch_reg)
    with h_control:
        for q in s0_qubits:
            emit_x(q)
    all_s0_with_h = [h_qubit_idx] + s0_qubits
    if len(all_s0_with_h) == 1:
        emit_z(all_s0_with_h[0])
    else:
        emit_mcz(all_s0_with_h[-1], all_s0_with_h[:-1])
    with h_control:
        for q in s0_qubits:
            emit_x(q)

    # --- Step 3c: U forward conditional on d(x) ---
    for d_val in range(1, d_max + 1):
        angle_data = self._variable_angles.get(d_val)
        if angle_data is None:
            continue
        phi_d = angle_data['phi']
        if is_root:
            phi_d = self._variable_root_angles.get(d_val, phi_d)
        cascade_ops_d = angle_data['cascade_ops']

        for pattern in itertools.combinations(range(d_max), d_val):
            zeros = [j for j in range(d_max) if j not in pattern]
            for z in zeros:
                emit_x(int(validity[z].qubits[63]))

            ctrl_qubits = [h_qubit_idx] + [int(validity[j].qubits[63]) for j in range(d_max)]

            # U forward: parent-child Ry then cascade
            _emit_multi_controlled_ry(
                self._height_qubit(depth - 1), phi_d, ctrl_qubits
            )
            if d_val > 1 and cascade_ops_d:
                _emit_cascade_multi_controlled(
                    branch_reg, cascade_ops_d, ctrl_qubits, sign=1
                )

            for z in zeros:
                emit_x(int(validity[z].qubits[63]))

    # --- Step 4: Uncompute validity ancillae ---
    # Reverse the evaluation loop (LIFO order: last child first)
    for i in reversed(range(d_max)):
        br = self.branch_registers[level_idx]
        bw = br.width
        for bit in range(bw):
            if (i >> (bw - 1 - bit)) & 1:
                emit_x(int(br.qubits[64 - bw + bit]))

        emit_x(self._height_qubit(depth))
        emit_x(self._height_qubit(depth - 1))

        node = self.node
        accept_q, reject_q = self._predicate(node)
        reject_qubit = int(reject_q.qubits[63])
        validity_qubit = int(validity[i].qubits[63])

        # Undo validity store (reverse order: X then CNOT)
        emit_x(validity_qubit)
        reject_ctrl = _make_qbool_wrapper(reject_qubit)
        with reject_ctrl:
            emit_x(validity_qubit)

        if self._predicate_is_compiled:
            self._predicate.adjoint(node)

        emit_x(self._height_qubit(depth - 1))
        emit_x(self._height_qubit(depth))

        for bit in range(bw):
            if (i >> (bw - 1 - bit)) & 1:
                emit_x(int(br.qubits[64 - bw + bit]))
```

**IMPORTANT SIMPLIFICATION:** The above pseudocode shows the full structure. However, the multi-controlled rotation logic with pattern matching becomes very complex for the executor to implement with proper gate decomposition. A SIMPLER and more practical approach is:

For binary branching (d_max=2 with 2 validity ancillae v0, v1):
- d=1, pattern (1,0): control on v0=|1>, v1=|0> -> X on v1, then controlled on h+v0+v1
- d=1, pattern (0,1): control on v0=|0>, v1=|1> -> X on v0, then controlled on h+v0+v1
- d=2, pattern (1,1): controlled on h+v0+v1 directly

For each pattern, the multi-controlled Ry can use the MCZ + Hadamard decomposition or emit individual controlled gates via the existing V-gate decomposition infrastructure.

**Add helper functions at module level (before the QWalkTree class):**

```python
def _emit_multi_controlled_ry(target_qubit, angle, ctrl_qubits):
    """Emit Ry(angle) on target controlled by multiple qubits.

    Uses the MCZ decomposition pattern: H(target) MCZ(target, ctrls) H(target)
    rotated to implement controlled-Ry rather than controlled-Z.

    For a single control, uses simple CRy (with qbool wrapper).
    For 2+ controls, decomposes via V-gate pattern.
    """
    from ._gates import emit_h, emit_mcz

    if not ctrl_qubits:
        emit_ry(target_qubit, angle)
        return

    if len(ctrl_qubits) == 1:
        ctrl = _make_qbool_wrapper(ctrl_qubits[0])
        with ctrl:
            emit_ry(target_qubit, angle)
        return

    # For 2 controls: V-gate decomposition CCRy
    if len(ctrl_qubits) == 2:
        c0, c1 = ctrl_qubits
        c0_ctrl = _make_qbool_wrapper(c0)
        c1_ctrl = _make_qbool_wrapper(c1)
        # CCRy(theta) = CRy(theta/2, c1) CNOT(c0,c1) CRy(-theta/2, c1) CNOT(c0,c1) CRy(theta/2, c0)
        with c1_ctrl:
            emit_ry(target_qubit, angle / 2)
        with c0_ctrl:
            emit_x(c1)
        with c1_ctrl:
            emit_ry(target_qubit, -angle / 2)
        with c0_ctrl:
            emit_x(c1)
        with c0_ctrl:
            emit_ry(target_qubit, angle / 2)
        return

    # For 3+ controls: recursive V-gate or raise
    # For d_max <= 3 we have at most 4 controls (h + 3 validity).
    # Use the H-MCZ-H decomposition for Ry:
    # CCC...Ry(theta) = Rz(-pi/2) H MCZ H Rz(pi/2) for rotations,
    # but simpler: use nested V-gate pattern.
    # Since d_max=3 means 4 controls max (h + v0 + v1 + v2),
    # decompose as: use last control as primary, rest as secondary.
    primary = ctrl_qubits[-1]
    remaining = ctrl_qubits[:-1]
    p_ctrl = _make_qbool_wrapper(primary)
    # V-gate: CRy(theta/2, primary) [anti-controlled block] CRy(theta/2, remaining-ctrl)
    with p_ctrl:
        emit_ry(target_qubit, angle / 2)
    # CNOT cascade from remaining to primary
    for r in remaining:
        r_ctrl = _make_qbool_wrapper(r)
        with r_ctrl:
            emit_x(primary)
    with p_ctrl:
        emit_ry(target_qubit, -angle / 2)
    for r in remaining:
        r_ctrl = _make_qbool_wrapper(r)
        with r_ctrl:
            emit_x(primary)
    # Remaining-controlled Ry(theta/2)
    _emit_multi_controlled_ry(target_qubit, angle / 2, remaining)
```

```python
def _emit_cascade_multi_controlled(branch_reg, ops, ctrl_qubits, sign=1):
    """Execute cascade ops with multiple control qubits.

    Each cascade gate operation is additionally controlled by all qubits
    in ctrl_qubits. Uses V-gate decomposition for multi-controlled gates.
    """
    w = branch_reg.width
    if sign == -1:
        ops = list(reversed(ops))

    for op in ops:
        if op[0] == 'ry':
            _, bit_off, angle = op
            qubit = int(branch_reg.qubits[64 - w + bit_off])
            _emit_multi_controlled_ry(qubit, sign * angle, ctrl_qubits)
        elif op[0] == 'cry':
            _, bit_off, angle, ctrl_bit_off = op
            qubit = int(branch_reg.qubits[64 - w + bit_off])
            cascade_ctrl = int(branch_reg.qubits[64 - w + ctrl_bit_off])
            _emit_multi_controlled_ry(qubit, sign * angle, ctrl_qubits + [cascade_ctrl])
        elif op[0] == 'x':
            _, bit_off = op
            qubit = int(branch_reg.qubits[64 - w + bit_off])
            # Multi-controlled X
            if not ctrl_qubits:
                emit_x(qubit)
            elif len(ctrl_qubits) == 1:
                ctrl = _make_qbool_wrapper(ctrl_qubits[0])
                with ctrl:
                    emit_x(qubit)
            else:
                # Multi-controlled X via H-MCZ-H
                from ._gates import emit_h, emit_mcz
                emit_h(qubit)
                emit_mcz(qubit, ctrl_qubits)
                emit_h(qubit)
        elif op[0] == 'cx':
            _, target_bit_off, ctrl_bit_off = op
            target_q = int(branch_reg.qubits[64 - w + target_bit_off])
            cascade_ctrl = int(branch_reg.qubits[64 - w + ctrl_bit_off])
            # Multi-controlled CX: control on cascade_ctrl + all ctrl_qubits
            all_ctrls = ctrl_qubits + [cascade_ctrl]
            from ._gates import emit_h, emit_mcz
            emit_h(target_q)
            emit_mcz(target_q, all_ctrls)
            emit_h(target_q)
```

**CRITICAL NOTES for the executor:**
1. The predicate evaluation loop navigates to each child by setting the branch register at `level_idx` (not `level_idx + 1`). The branch register at `level_idx` stores which child of the node at depth `depth` was taken. Setting it to child index `i` represents navigating to that child.
2. Height flip moves from h[depth] to h[depth-1] (going down one level to where the child lives).
