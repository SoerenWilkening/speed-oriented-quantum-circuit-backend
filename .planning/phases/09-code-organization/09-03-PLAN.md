   ```c
   // Qubit layout: [0, bits-1] = target operand
   sequence_t *Q_not(int bits);   // Parallel X gates (O(1) depth)
   sequence_t *cQ_not(int bits);  // Controlled NOT with sequential CX (O(bits) depth)

   // Width-parameterized XOR operations
   // Qubit layout: [0, bits-1] = target A, [bits, 2*bits-1] = operand B
   sequence_t *Q_xor(int bits);   // Parallel CNOT (O(1) depth)
   sequence_t *cQ_xor(int bits);  // Controlled XOR with Toffoli (O(bits) depth)

   // Width-parameterized AND operations
   // Qubit layout: [0, bits-1] = result, [bits, 2*bits-1] = A, [2*bits, 3*bits-1] = B
   sequence_t *Q_and(int bits);   // Parallel Toffoli (O(1) depth)
   sequence_t *CQ_and(int bits, int64_t value);  // Classical-quantum AND

   // Width-parameterized OR operations
   // A OR B = A XOR B XOR (A AND B)
   // Qubit layout: same as AND
   sequence_t *Q_or(int bits);    // 3-layer circuit
   sequence_t *CQ_or(int bits, int64_t value);   // Classical-quantum OR
   ```

5. Add OWNERSHIP comments: "Returns cached sequence - DO NOT FREE" where applicable
