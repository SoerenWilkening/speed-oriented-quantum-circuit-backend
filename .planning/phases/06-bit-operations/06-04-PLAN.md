    """Phase 6: Bitwise Operations Tests

    Tests for BITOP-01 through BITOP-05 requirements:
    - BITOP-01: Bitwise AND on quantum integers
    - BITOP-02: Bitwise OR on quantum integers
    - BITOP-03: Bitwise XOR on quantum integers
    - BITOP-04: Bitwise NOT on quantum integers
    - BITOP-05: Python operator overloading for bitwise ops
    """
    import pytest
    from quantum_language import qint, qbool, circuit


    class TestBitopAnd:
        """Tests for BITOP-01: Bitwise AND"""

        def test_and_same_width(self):
            """AND of two same-width qints produces same-width result"""
            a = qint(0b1010, width=4)
            b = qint(0b1100, width=4)
            c = a & b
            assert c.width == 4
            # Circuit should have Toffoli gates

        def test_and_mixed_width(self):
            """AND of different widths uses larger width"""
            a = qint(0xFF, width=8)
            b = qint(0xFFFF, width=16)
            c = a & b
            assert c.width == 16

        def test_and_qint_int(self):
            """AND with classical integer"""
            a = qint(0b1111, width=4)
            c = a & 0b1010
            assert c.width == 4

        def test_and_preserves_operands(self):
            """AND does not modify input operands"""
            a = qint(5, width=8)
            b = qint(3, width=8)
            a_qubits_before = list(a.qubits)
            b_qubits_before = list(b.qubits)
            c = a & b
            assert list(a.qubits) == a_qubits_before
            assert list(b.qubits) == b_qubits_before


    class TestBitopOr:
        """Tests for BITOP-02: Bitwise OR"""

        def test_or_same_width(self):
            """OR of two same-width qints produces same-width result"""
            a = qint(0b1010, width=4)
            b = qint(0b0101, width=4)
            c = a | b
            assert c.width == 4

        def test_or_mixed_width(self):
            """OR of different widths uses larger width"""
            a = qint(0x0F, width=8)
            b = qint(0xF000, width=16)
            c = a | b
            assert c.width == 16

        def test_or_qint_int(self):
            """OR with classical integer"""
            a = qint(0b0000, width=4)
            c = a | 0b1111
            assert c.width == 4

        def test_or_preserves_operands(self):
            """OR does not modify input operands"""
            a = qint(5, width=8)
            b = qint(3, width=8)
            a_qubits_before = list(a.qubits)
            b_qubits_before = list(b.qubits)
            c = a | b
            assert list(a.qubits) == a_qubits_before
            assert list(b.qubits) == b_qubits_before


    class TestBitopXor:
        """Tests for BITOP-03: Bitwise XOR"""

        def test_xor_same_width(self):
            """XOR of two same-width qints produces same-width result"""
            a = qint(0b1010, width=4)
            b = qint(0b1100, width=4)
            c = a ^ b
            assert c.width == 4

        def test_xor_mixed_width(self):
            """XOR of different widths uses larger width"""
            a = qint(0xFF, width=8)
            b = qint(0x00FF, width=16)
            c = a ^ b
            assert c.width == 16

        def test_xor_qint_int(self):
            """XOR with classical integer"""
            a = qint(0b1111, width=4)
            c = a ^ 0b1010
            assert c.width == 4

        def test_xor_preserves_operands(self):
            """XOR does not modify input operands"""
            a = qint(5, width=8)
            b = qint(3, width=8)
            a_qubits_before = list(a.qubits)
            b_qubits_before = list(b.qubits)
            c = a ^ b
            assert list(a.qubits) == a_qubits_before
            assert list(b.qubits) == b_qubits_before


    class TestBitopNot:
        """Tests for BITOP-04: Bitwise NOT"""

        def test_not_same_width(self):
            """NOT preserves width"""
            a = qint(0b1010, width=4)
            b = ~a
            assert b.width == 4

        def test_not_inplace(self):
            """NOT modifies in-place and returns self"""
            a = qint(0b0000, width=4)
            b = ~a
            assert b is a  # Same object

        def test_not_qbool(self):
            """NOT works on qbool (1-bit)"""
            a = qbool(False)
            b = ~a
            assert b.width == 1
            assert b is a


    class TestBitopOverloading:
        """Tests for BITOP-05: Python operator overloading"""

        def test_and_operator(self):
            """& operator works"""
            a = qint(1, width=8)
            b = qint(1, width=8)
            c = a & b
            assert isinstance(c, qint)

        def test_or_operator(self):
            """| operator works"""
            a = qint(1, width=8)
            b = qint(1, width=8)
            c = a | b
            assert isinstance(c, qint)

        def test_xor_operator(self):
            """^ operator works"""
            a = qint(1, width=8)
            b = qint(1, width=8)
            c = a ^ b
            assert isinstance(c, qint)

        def test_invert_operator(self):
            """~ operator works"""
            a = qint(1, width=8)
            b = ~a
            assert isinstance(b, qint)

        def test_iand_operator(self):
            """&= operator works"""
            a = qint(1, width=8)
            b = qint(1, width=8)
            a &= b
            assert isinstance(a, qint)

        def test_ior_operator(self):
            """|= operator works"""
            a = qint(1, width=8)
            b = qint(1, width=8)
            a |= b
            assert isinstance(a, qint)

        def test_ixor_operator(self):
            """^= operator works"""
            a = qint(1, width=8)
            b = qint(1, width=8)
            a ^= b
            assert isinstance(a, qint)


    class TestPhase6SuccessCriteria:
        """Tests mapping to Phase 6 success criteria from ROADMAP.md

        Success Criteria:
        1. Bitwise AND, OR, XOR, and NOT operations work on quantum integers
        2. Python operator overloading works for bitwise operations
        3. Bit operations respect variable-width integers
        4. Generated circuits for bit operations have reasonable depth
        """

        def test_criterion_1_all_bitwise_ops_work(self):
            """SC1: AND, OR, XOR, NOT all work on qint"""
            a = qint(0b1010, width=4)
            b = qint(0b1100, width=4)

            # All operations should execute without error
            c_and = a & b
            c_or = a | b
            c_xor = a ^ b
            c_not = ~a

            assert isinstance(c_and, qint)
            assert isinstance(c_or, qint)
            assert isinstance(c_xor, qint)
            assert isinstance(c_not, qint)

        def test_criterion_2_operator_overloading(self):
            """SC2: Python operators &, |, ^, ~ work"""
            a = qint(5, width=8)
            b = qint(3, width=8)

            # Using Python operators directly
            _ = a & b
            _ = a | b
            _ = a ^ b
            _ = ~a

        def test_criterion_3_variable_width_respected(self):
            """SC3: Operations respect variable-width"""
            a = qint(0, width=8)
            b = qint(0, width=16)
            c = qint(0, width=32)

            # Result should use max width
            r1 = a & b
            r2 = b | c
            r3 = a ^ c

            assert r1.width == 16
            assert r2.width == 32
            assert r3.width == 32

        def test_criterion_4_reasonable_depth(self):
            """SC4: Circuit depth is reasonable (O(N) or better)

            NOT and XOR should be O(1) depth (parallel gates)
            AND and OR should be O(N) depth (sequential Toffoli)
            """
            # This is a characterization test - circuits should complete
            # without timeout for reasonable sizes
            a = qint(0, width=32)
            b = qint(0, width=32)

            # These should complete quickly
            _ = ~a
            _ = a ^ b
            _ = a & b
            _ = a | b


    class TestBackwardCompatibility:
        """Tests ensuring qbool bitwise ops still work"""

        def test_qbool_and(self):
            """qbool & qbool still works"""
            a = qbool(True)
            b = qbool(True)
            c = a & b
            assert isinstance(c, (qint, qbool))
            assert c.width == 1

        def test_qbool_or(self):
            """qbool | qbool still works"""
            a = qbool(True)
            b = qbool(False)
            c = a | b
            assert isinstance(c, (qint, qbool))
            assert c.width == 1

        def test_qbool_not(self):
            """~qbool still works"""
            a = qbool(True)
            b = ~a
            assert isinstance(b, (qint, qbool))
            assert b.width == 1
